int partition(arrI base, int i, int m, int pivot)
{
  if(i==m){
    return i;
  }
  else{
    // P(base,i,m) |- base+i::arrI<_> * R1(base,i,m)
    int tmpi=base[i];

    // (R1(base,i,m) |- base+m-1::arrI<_> * R3(base,i,m))

      int tmp1 = partition(base,i,m-1,pivot);
      return tmp1;
     }

    int tmp2=partition(base,i+1,m,pivot);
